import java.util.*;

class Solution {
    public List<List<String>> groupOfWords(String[] strs) {
        if (strs.length == 0) return new ArrayList();
        Map<String, List> ans = new LinkedHashMap<String, List>();
        for (String s : strs) {
            char[] ca = s.toCharArray();
            Arrays.sort(ca);
            String key = String.valueOf(ca);
            if (!ans.containsKey(key)) ans.put(key, new ArrayList());
            ans.get(key).add(s);
        }
        return new ArrayList(ans.values());
    }

    public static void main(String args[]) // throws IOException 
    {
        Scanner sc = new Scanner(System.in);
		String str =sc.nextLine();
		String[] list = str.split(" ");
        Solution fna=new Solution();//Fenwick Tree Approch Class Object Creation
        System.out.println(fna.groupOfWords(list));//call Fenwick sumRange()
    }
}
